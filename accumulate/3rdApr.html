      	//准备数据
		var nodes = [
        { name: "航空航天" ,id:0},

        var edges = [
        {id:0, source: 0, target: 1, relation: "航空航天", value: .3,type:"resolved" },
        {id:1,  source: 0, target: 2, relation: "航空航天", value: 1 ,type:"resolved" },
        {id:2,  source: 0, target: 3, relation: "航空航天", value: 1 ,type:"resolved"},

		{id:4,  source: 0, target: 5, relation: "航空航天", value: 1.3 ,type:"resolved"},
        {id:5,  source: 0, target: 6, relation: "航空航天", value: 1 ,type:"resolved"},
        {id:6,  source: 0, target: 7, relation: "航空航天", value: 1 ,type:"resolved"},
        {id:7,  source: 0, target: 8, relation: "航空航天", value: 1 ,type:"resolved"},
		{id:8,  source: 1, target: 9, relation: "航空航天", value: 1 ,type:"resolved"},
        {id:9,  source: 1, target: 10, relation: "航空航天", value: 1 ,type:"resolved"},
		{id:10,  source: 1, target: 11, relation: "航空航天", value: 1 ,type:"resolved"},
        ];
		var nodesMap={};
		var linkMap={};
		// 生成 nodes map
		nodesMap = genNodesMap(nodes);
		// 构建 nodes（不能直接使用 api 中的 nodes）
		nodes = d3.values(nodesMap);
		// 起点和终点相同的关系映射
		linkMap = genLinkMap(edges);
		// 构建 links（source 属性必须从 0 开始）
		var links = genLinks(edges);
      	//设置一个color的颜色比例尺，为了让不同的扇形呈现不同的颜色
		var colorScale = d3
			.scaleOrdinal()
			.domain(d3.range(nodes.length))
			.range(d3.schemeCategory10);		
		//新建一个力导向图
		var forceSimulation = d3
			.forceSimulation()
			
			.force("link", d3.forceLink())
			.force("charge", d3.forceManyBody())
			.force("center", d3.forceCenter());		
		//初始化力导向图，也就是传入数据
		//生成节点数据
		forceSimulation.nodes(nodes).on("tick", ticked); 
		//生成边数据
		forceSimulation
			.force("link")
			.links(edges)
			.distance(function(d) {
		    //每一边的长度
		    return d.value * 200;
			});
		//设置图形的中心位置
		forceSimulation
			.force("center")
			.x(width / 2)
			.y(height / 2);
		//有了节点和边的数据后，我们开始绘制
		//绘制箭头
		var marker = svg
			.append("marker")
			.attr("id", "arrow")
			.attr("markerUnits", "strokeWidth")
			.attr("viewBox", "0 -5 10 10")
			.attr("refX", 32)
			.attr("refY", -1)
			.attr("markerWidth", 12)
			.attr("markerHeight", 12)
			.attr("orient", "auto")
			.attr("stroke-width", 2)
			.append("path")
			.attr("d", "M0,-5L10,0L0,5")
			.attr("fill", "#808080");	
		//绘制边
		links = g
			.append("g")
			.selectAll("line")
			.data(edges)
			.enter()
			.append("line")
			.attr("stroke", function(d, i) {
			return colorScale(i);
			})
			.attr("stroke-width", 1)
			.attr("class",'link')
			.attr('id','link-' + links.id)
			.attr("marker-end", "url(#arrow)");
		var linksText = g
			.append("g")
			.selectAll("text")
			.data(edges)
			.enter()
			.append("text")
			.style("pointer-events", "none")
			.text(function(d) {
			return d.relation;
			});		
		//绘制节点
		//先为节点和节点上的文字分组
		var gs = g
			.selectAll(".circleText")
			.data(nodes)
			.enter()
			.append("g")
			.attr("transform", function(d, i) {
				var cirX = d.x;
				var cirY = d.y;
				return "translate(" + cirX + "," + cirY + ")";
			})
			.call(
				d3
					.drag()
					.on("start", started)
					.on("drag", dragged)
					.on("end", ended)
            );
        //绘制浮动提示窗口
		var tooltip=d3.select("body")
			.append("div")
			.attr("class","tooltip")
			.style("opacity",0.0);
      	//绘制节点
		var nodeCircle = gs.append("circle")
        .attr("r", 30)
        .attr("fill", function(d, i) {
			return colorScale(i);
		})
		.attr('class', 'node')
		.attr('class', 'node-circle')
		.attr('id','node-circle-' + nodes.id)
		.on("mouseout",function(currNode,d){//移出
			if (draging) {
                return;
            }
			var g = d3.select(this)
			.transition()
			.duration(1500)
			.attr("r","30");
			toggleNode(nodeCircle, currNode, false);
            toggleLine(links, currNode, false);
            toggleMarker(marker, currNode, false);
			toggleLineText(linksText, currNode, false);
			tooltip.style("opacity",0);
		})
		.on("mouseover",function(currNode,d){//移入
			if(draging){
				return;
			}
			var g = d3.select(this)
			.transition()
			.duration(1500)
			.attr("r","40");
			tooltip.html( "信息" + "<br />" +  "再加点信息")
			.style("left", (d3.event.pageX) + "px")
			.style("top", (d3.event.pageY + 20) + "px")
			.style("opacity",1.0);
			console.log(111)
			toggleLineText(linksText, currNode, true);
			toggleNode(nodeCircle, currNode, true);
            toggleLine(links, currNode, true);
			toggleMarker(marker, currNode, true);
		})
		.on("mousemove",function(d){ //在上边移动
			tooltip.style("left", (d3.event.pageX) + "px")
			.style("top", (d3.event.pageY + 20) + "px");
		})
		;
      	//文字
		gs.append("text")
		.attr("dy", ".35em")
		.style("pointer-events", "none")
        .text(function(d) {
			return d.name;
		})
		.attr('text-anchor', 'middle');
		gs.append("circle".c)
		//ticked
		function ticked() {
        links
			.attr("x1", function(d) {
            return d.source.x;
			})
			.attr("y1", function(d) {
            return d.source.y;
			})
			.attr("x2", function(d) {
            return d.target.x;
			})
			.attr("y2", function(d) {
            return d.target.y;
			});
        linksText
			.attr("x", function(d) {
            return (d.source.x + d.target.x) / 2;
			})
			.attr("y", function(d) {
            return (d.source.y + d.target.y) / 2;
		});

        gs.attr("transform", function(d) {
			return "translate(" + d.x + "," + d.y + ")";
        });
		}
		//drag
		function started(d) {
			draging = true;
			if (!d3.event.active) {
				forceSimulation.alphaTarget(0.8).restart();
			}
			d.fx = d.x;
			d.fy = d.y;
		}
		function dragged(d) {
			draging = true;
			d.fx = d3.event.x;
			d.fy = d3.event.y;
		}
		function ended(d) {
			draging= false;
			if (!d3.event.active) {
				forceSimulation.alphaTarget(0);
			}	
			d.fx = null;
			d.fy = null;
        }
        //links node - map初始化
		function genLinks(edges) {
			var indexHash = {};
			return edges.map(function ({
				id,
				source,
				target,
				relation,
				type
			}, i) 
			{
				var linkKey = source + '-' + target;
				var count = linkMap[linkKey];
				if (indexHash[linkKey]) {
					indexHash[linkKey] -= 1;
				} else {
					indexHash[linkKey] = count - 1;
				}
			
				return {
					id,
					source,
					target,
					relation,
					type,
					labels: linkMap[linkKey + '-relation'],
					count: linkMap[linkKey],
					index: indexHash[linkKey]
				}
			})
		}

		function genLinkMap(edges) {
			const hash = {};
			edges.map(function ({
				source,
				target,
				relation
			}) 
			{
				const key = source + '-' + target;
				if (hash[key]) {
					hash[key] += 1;
					hash[key + '-relation'] += '、' + relation;
				} else {
					hash[key] = 1;
					hash[key + '-relation'] = relation;
				}
			});	
		return hash;
		}
		function genNodesMap(nodes) {
			const hash = {};
			nodes.map(function ({
				id,
				name,
				}) 
			{
				hash[id] = {
					id,
					name,
				};
			});
		return hash;
		}
		//确认哪条线
		function isLinkLine(node, links) {
			var flag1 = (links.source.id == node.id);
			var flag2 = (links.target.id == node.id);
			return flag1 || flag2;
		}
		function isLinkNode(currNode, node) {
			if (currNode.id === node.id) {
				return true;
			}
			return linkMap[currNode.id + '-' + node.id] || linkMap[node.id + '-' + currNode.id];
		}
		//mouseover
		function toggleNode(nodeCircle, currNode, isHover) {
		if (isHover) {
			// this.parentNode.appendChild(this);
			nodeCircle
				.style('opacity', .1)
				.filter(node => isLinkNode(currNode, node))
				.style('opacity', 1);
			} else {
				nodeCircle.style('opacity', 1);
			}
		
		}
		function toggleLine(links, currNode, isHover) {
			if (isHover) {
		        // 加重连线样式
				links
					.style('opacity', .1)
					.filter(links => isLinkLine(currNode, links))
					.style('opacity', 1)
					.classed('link-active', true);
			} else {
		        // 连线恢复样式
				links
					.style('opacity', 1)
					.classed('link-active', false);
			}
		}
		function toggleLineText(linksText, currNode, isHover) {
			if (isHover) {
		        // 只显示相连连线文字
				linksText
				.style('fill-opacity', .1)
				.filter(links => isLinkLine(currNode, links))
				.style('fill-opacity', 1);
			} else {
		        // 显示所有连线文字
				linksText
					.style('fill-opacity', '1.0');
			}
		}
		function toggleMarker(marker, currNode, isHover) {
			if (isHover) {
		        // 放大箭头
				marker.filter(links => isLinkLine(currNode, links))
					.style('transform', 'scale(1.5)');
			} else {
		        // 恢复箭头
				marker
					.style('transform', 'scale(1)');
			}
		}
