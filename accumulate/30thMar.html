# 基础函数

# 选择元素

主要就是两个很好理解的函数,select()和selectAll()

    var example1 = d3.select("body"); //选择
    var example2 = d3.select("body").selectAll("p"); //选择body下的一类所有的标签,比如p
    example2.text();//这样就能改变p的内容了

如果不是selectAll(p),而是selecet(p)，那么会自动选择第一个元素。

个体选择就是和jquery选择元素一样。select(.class)和select(#id)。

append()：在选择集尾部插入元素。var p = d3.select("body").append("p")

insert()：在选择集前面插入元素。var p = d3.select("body").insert("p","#myP3")。insert("p","#myP3")表示在属性id为myP3的元素前面插入一个新的元素<p>。

remove()：对于选择的元素进行删除。var p = d3.select("body").select("#myp3").remove().

# 数据绑定

data()：讲一个数组绑定到选择集上，数组各项和选择集各元素绑定.

datum()：将一个数据绑定到所有选择集上.

区别就是data是一一对应的，所以一般data比较常用。

<body>
    <p>dog</p>
    <p>cat</p>
    <p>pig</p>
    <script>
    	var str = "is an animal";//新建一个字符串,只对这个字符串进行操作
    	var p = d3.select("body")
    		.selectAll("p");
    	p.datum(str)//绑定
    		.text(function(d,i){
    			return "第"+i+"个元素"+d;//第1个动物is an animal;第2个动物is an animal;
    		});
    </script>
  </body>

# Update和Enter

例子一：update和enter：数组[3,6,9,12,15]绑定到三个

上。可以想象到，数组的最后两个元素没有可以绑定的元素，这时D3会建立两个空的元素与数组最后的两个数据相对于，那么这部分就称为Enter。而有元素与数据对应的部分就称为Update

一共p p p,- - - - -；(p- p- p-)Update (空- 空-)Enter

<body>
    <p>dog</p>
    <p>cat</p>
    <p>pig</p>
    <script>
    	var dataset = [3,6,9,12,15];
    	var p = d3.select("body")
    		.selectAll("p");
    	var update = p.data(dataset)// 表示绑定数据,并得到update部分
    	var enter = update.enter();// 表示得到enter部分
    	//下面检验是否真的得到
    	//对于update的处理
    	update.text(function(d,i){
    		return "update: "+d+",index: "+i;
        })
        
    	//对于enter的处理
    	//注意，这里需要先添加足够多的<p>，然后在添加文本
    	var pEnter = enter.append("p")//添加足够多的<p>
    	pEnter.text(function(d,i){
    		return "enter: "+d+",index: "+i;
    	})
    </script>
  </body>

# Update和Exit

例子二：exit：如果数组[3]绑定到三个

上，可以想象，最后两个

没有可绑定的数据，那么没有数据绑定的部分就称为Exit

一共p p p,-；(p-)Update (p无 p无)Exit

<body>
    <p>dog</p>
    <p>cat</p>
    <p>pig</p>
    <p>rat</p>
    <script>
    	var dataset = [3,6];
    	var p = d3.select("body")
    		.selectAll("p");

    	var exit = update.exit();//得到exit部分
    	//下面检验是否真的得到
    	//对于update的处理
    	update.text(function(d,i){
    		return "update: "+d+",index: "+i;
        })
        //对于exit的处理
    	exit.remove();  	
    	exit.text(function(d,i){
    		return "exit";
    	})
    </script>

# 绘制图表

# 简单图表

绘制一个D3.js的简单图表，除了那些基础函数，我们还需要学习一下几点

  • svg画布：svg绘制的是矢量图（还有canvas画布，这个是JavaScript用来绘制2D图像的，是位图）
  • rect元素：是d3中在svg中绘制矩形的元素
  • g元素：分组的时候使用

接下来就可以开始我们的第一个简单图表的绘制了。

var marge = {top:60,bottom:60,left:60,right:60}//设置边距
var dataset = [ 250 , 210 , 170 , 130 , 90 ];  //数据（表示矩形的宽度）
var svg = d3.select("svg");//得到svg画布
var g  = svg.append("g")//定义一个用来装整个图表的一个分组，并设置他的位置
            .attr("transform","translate("+marge.top+","+marge.left+")");
            //.attr(attribute,value/function(index,oldvalue)); attribute是规定属性的名称；value或者function不用解释。
            //详情可查找jquery手册
            //transform()属性向元素应用 2D 或 3D 转换。该属性允许我们对元素进行旋转、缩放、移动或倾斜。
            //translate进行的是2D转换。
            //详情可查找css3手册

# 动起来

  • .attr(xxx) .transition() 表示添加过渡，也就是从前一个属性过渡到后一个属性
  • .duration(2000)，表示过渡时间持续2秒
  • .delay(500)，表示延迟0.4秒后再进行过渡
  • .ease(d3.easeElasticInOut)表示过渡方式，注意这里和v3版本的区别

有了上面的基础知识后，现在我们来实现动态图表

# 为图像添加动态效果

gs.append("rect")
    		.attr("x",function(d,i){
    			return xScale(i)+rectPadding/2; //比例尺
    		})	
    		.attr("y",function(d){//这里是要改变的，即初始状态
    			var min = yScale.domain()[0];
    			return yScale(min);//可以得知，这里返回的是最大值
    		})
    		.attr("width",function(){
    			return xScale.step()-rectPadding;
    		})
    		.attr("height",function(d){//这里要改变，即初始状态
    			return 0;
    		})
    		.attr("fill","blue")
    		.transition()//启动过渡效果。其前后是图形变化前后的状态（形状、位置、颜色等等）
    		.duration(2000)//指定过渡的持续时间，单位为毫秒。
    		.delay(function(d,i){//表示一定时间后才开始转变，单位同样为毫秒。此函数可以对整体指定延迟，也可以对个别指定延迟。
    			return i*400;
    		})
			//.ease(d3.easeElasticInOut)//这里读者可以自己将注释去掉，看看效果（chrome浏览器会报错，但是不影响效果）
			//指定过渡的变化方式,inear：普通的线性变化；circle：慢慢地到达变换的最终状态；
			//elastic：带有弹跳的到达最终状态；bounce：在最终状态处弹跳几次；
    		.attr("y",function(d){//回到最终状态
    			return yScale(d);
    		})
    		.attr("height",function(d){//回到最终状态
    			return height-marge.top-marge.bottom-yScale(d);
    		})

gs.append("text")
    		.attr("x",function(d,i){
    			return xScale(i)+rectPadding/2;
    		})
    		.attr("y",function(d){
            	var min = yScale.domain()[0];

        		(xScale.step()-rectPadding)/2;
