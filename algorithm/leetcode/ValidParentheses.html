# Valid Parentheses

Given a string containing just the characters '(', ')', '{', '}', '[' and ']', determine if the input string is valid.

An input string is valid if:

Open brackets must be closed by the same type of brackets.

Open brackets must be closed in the correct order.

Note that an empty string is also considered valid.

# 我的解

def isValid(self, s: str) -> bool:
    map_a = {'(':1,'{':2,'[':3,')':4,'}':5,']':6}
    map_b = {')':1,'}':2,']':3}
    flag = 0
    if len(s) == 1:
        return False
    for i,j in enumerate(s):
        if j not in map_b:
            flag += 1
        if j in map_b:
            flag -=1
            if map_a[s[i-1]] != map_b[j] and s[i-1] not in map_b:
                return False
    if flag == 0:
        return True
    else: return False

# 最优解

def isValid(self, s: str) -> bool:
    op = '({['
    cp = ')}]'
    cp_to_op = {')': '(',
               '}': '{',
               ']': '['}
    paren = list()
    
    for c in s:
        if c in op:
            paren.append(c)
        elif c in cp:
            if len(paren) == 0:
                return False
            if cp_to_op[c] != paren.pop():
                return False
    if len(paren) > 0:
        return False
    else:
        return True
